TempusJS demo

This is demo examples of TempusJS.

Here timestamps in seconds! In default JS in milliseconds!

Format

Directive Meaning Example
d Day of the month as a zero-padded decimal number. 01, 02, ..., 31
m Month as a zero-padded decimal number. 01, 02, ..., 12
Y Year with century as a decimal number. 1970, 1988, 2001, 2013
w Weekday as a decimal number, where 0 is Sunday and 6 is Saturday. 0, 1, ..., 6
a Weekday abbreviated name. Sun, Mon, ..., Sat
A Weekday full name. Sunday, Monday, ..., Saturday
b Month abbreviated name. Jan, Feb, ..., Dec
B Month full name. January, February, ..., December
H Hour (24-hour clock) as a zero-padded decimal number. 00, 01, ..., 23
M Minute as a zero-padded decimal number. 00, 01, ..., 59
S Second as a zero-padded decimal number. 00, 01, ..., 59
s Unix time (in seconds). 0, ..., 1377157851, ..., 2147483647
F ISO_8601 format (%Y-%m-%d). 1970-01-01, ..., 2013-08-22, ...
D UK and USA format (%m/%d/%Y). 01/01/1970, ..., 08/22/2013, ...

Methods

int time()
Return current timestamp (UTC) in seconds. If parameter is not undefined, timestamp was received from this. 
// t = 1383667313;
var t = tempus.time();
Run 
// t = 1383609600;
var t = tempus.time({year: 2013, month: 11, day: 5});
Run 
object now( [string format] )
Return current date as object or formatted string. 
//  t = {
//    day: 21,
//    dayOfWeek: 1,
//    hours: 15,
//    minutes: 10,
//    month: 10,
//    seconds: 3,
//    timestamp: 1382353803,
//    year: 2013
//  };
var t = tempus.now();
Run 
//  t = '21.10.2013';
var t = tempus.now('d.m.Y');
Run 
bool isLeapYear( [int year] )
If year is leap return true. If no parameters - checking current year. 
//  checking leap of current year.
var leapYear = tempus.isLeapYear();
Run 
//  leapYear = true;
var leapYear = tempus.isLeapYear(2004);
Run

tempus.format (date, format)

Syntax

string format( object|int date, string format )

Description

Return formatted string of date. You can use object or timestamp as parameter of method.

Examples

//  t = '05.11.2013';
var t = tempus.format({year: 2013, month: 11, day:5}, 'd.m.Y');
Run 
//  t = '2013-11-05 12:36:42';
var t = tempus.format(tempus.now(), 'Y-m-d H:M:S');
Run 
//  t = '20131105';
var t = tempus.format(tempus.time(), 'Ymd');
Run

tempus.getDaysCountInMonth (month, year)

Syntax

int getDaysCountInMonth( int month, [int year] )

Description

Return days count in month. If year is undefined - leap year is false.

Examples

Run 
//  t = 29;

Run 
//  t = 28;
var t = tempus.getDaysCountInMonth(2);
Run

tempus.getMonthNames (longNames)

Syntax

object getMonthNames( [bool longNames] )

Description

Return array of month names. If longNames is undefined, short names was returned.

Examples

//  t = ["jan","feb","mar","apr","may","jun","jul","aug","sep","oct","nov","dec"];
var t = tempus.getMonthNames();
Run 
//  t = ["january","february","march","april","may","june",
//       "july","august","september","october","november","december"];
var t = tempus.getMonthNames(true);
Run

tempus.incDate (date, value, type)

Syntax

object incDate( object date, int value, string type=day|month|year )

Description

Return date object, increased on [value] [type].

Examples

//  t = {"year":2013,"month":10,"day":12};
var t = tempus.incDate({year: 2013, month: 10, day: 5}, 7, 'day');
Run 
//  t = {"year":2014,"month":1,"day":13};
var t = tempus.incDate({year: 2013, month: 10, day: 25}, 80, 'day');
Run 
//  t = {"year":2013,"month":12,"day":30};
var t = tempus.incDate({year: 2013, month: 1, day: 30}, 11, 'month');
Run 
//  t = {"year":2015,"month":1,"day":1};
var t = tempus.incDate({year: 2000, month: 1, day: 1}, 15, 'year');
Run

tempus.between (dateFrom, dateTo, type)

Syntax

int between( object dateFrom, object dateTo, string type=day|month|year|hours|minutes|seconds )

Description

Return integer of date between from [dateFrom] to [dateTo] as [type].

Examples

//  t = 4;
var t = tempus.between({year: 2013, month: 11, day: 1}, {year: 2013, month: 11, day: 5}, 'day');
Run 
//  t = 6;
var t = tempus.between({year: 2013, month: 11, day: 1}, {year: 2014, month: 5, day: 5}, 'month');
Run 
//  t = 266400;
var t = tempus.between({year: 2013, month: 11, day: 1}, {year: 2014, month: 5, day: 5}, 'minutes');
Run 
//  t = 10224;
var t = tempus.between({year: 2013, month: 11, day: 1}, {year: 2015, month: 1, day: 1}, 'hours');
Run 
//  Happy new year!
//  t = 56;
var t = tempus.between(tempus.now(), {year: 2014, month: 1, day: 1}, 'day');
Run

Getting current datetime - all formatting

Run 
// Example result:
// 2013-03-14 4 3 Thu Thursday Mar March 14:57:01 1363258621 2013-03-14 03/14/2013
// format like here: http://php.net/manual/function.strftime.php
var d = $.datetime('%Y-%m-%d %w %a %A %b %B %H:%M:%S %s %F %D');

Format datetime

Run 
// Example result:
// 12.03.2013
var d = $.datetime.format('%d.%m.%Y', 1363046400);
Run 
// Example result:
// Tue 12, 2013
var d = $.datetime.format('%a %d, %Y', 1363046400);

Reformat datetime

Run 
// Example result:
// 12.03.2013
var d = $.datetime.reformat('2013-03-12', '%d.%m.%Y');
Run 
// Example result:
// 15.03.2013
var d = $.datetime.reformat('2013-03-15 16:00', '%d.%m.%Y', '%Y-%m-%d %H:%M');
Run 
// Example result:
// 12:31 (Fri 15, 2013)
var d = $.datetime.reformat('2013-03-15T12:31:48', '%H:%M (%a %d, %Y)', '%Y-%m-%dT%H:%M:%S');

String to timestamp [+ by format]

Run 
// Example result:
// 1363046400
var d = $.datetime.strtotime('2013-03-12');
Run 
// Example result:
// 1363348800
var d = $.datetime.strtotime('2013-03-15 16:00', '%Y-%m-%d %H:%M');

Validate date/time by format

Run 
// Example result:
// true
var d = $.datetime.validate('2013-03-12', '%Y-%m-%d');
Run 
// Example result:
// true
var d = $.datetime.validate('16:00 08.08.2013', '%H:%M %d.%m.%Y');
Run 
// Example result:
// false
var d = $.datetime.validate('32.08.2013', '%d.%m.%Y');
Run 
// Example result:
// false
var d = $.datetime.validate('29.02.2013', '%d.%m.%Y');
Run 
// Example result (Leap year):
// true
var d = $.datetime.validate('29.02.2012', '%d.%m.%Y');
Run 
// Example result (Leap year):
// false
var d = $.datetime.validate('24:61 29.02.2012', '%H:%M %d.%m.%Y');
Run 
// Example result:
// true
var d = $.datetime.validate('00:00 01.01.2012', '%H:%M %d.%m.%Y');

Validate date/time by format with ranges

Run 
// Example result:
// true
var d = $.datetime.validate('08.08.2013', '%d.%m.%Y', {year: {min:1970, max: 2038}});
Run 
// Example result:
// false
var d = $.datetime.validate('08.08.1961', '%d.%m.%Y', {year: {min:1970, max: 2038}});

allow_empty param

Run 
// Example result:
// true
var d = $.datetime.validate('', '%d.%m.%Y', {year: {min:1970, max: 2038}}, true);
