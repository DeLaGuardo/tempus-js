tempus.format (date, format)

Syntax

string format( object|int date, string format )

Description

Returns formatted string of date. You can use object or timestamp as parameter of method.

Examples

//  t = '05.11.2013';
var t = tempus.format({year: 2013, month: 11, day:5}, '%d.%m.%Y');
Run 
//  t = '2013-11-05 12:36:42';
var t = tempus.format(tempus.now(), '%Y-%m-%d %H:%M:%S');
Run 
//  t = '20131105';
var t = tempus.format(tempus.time(), '%Y%m%d');
Run

tempus.between (dateFrom, dateTo, type)

tempus.clock (callback)

Syntax

object clock( function callback )

Description

Clock. Every second callback is calling with parameter "date". "date" - is current date object. Also run callback immediately after starting.

Examples

//  For example, please, click button
var clock = tempus.clock(function(date) {
    document.getElementById('tempus-clock-example').innerHTML = tempus.format(date, "%H:%M:%S");
});

tempus.generateDates (date, formatFrom, formatTo)

Syntax

string generateDates( object options )

Options:

{
    dateFrom: string|object
    formatFrom: string|undefined
    dateTo: string|object
    formatTo: string|undefined
    period: number (seconds)|object|string='seconds'|'minutes'|'hours'|'day'|'month'|'year'
    format: string|undefined,
    asObject: boolean,
    groupBy: string=year|month|week|day|hours|minutes|seconds
}

Description

Generates dates from [dateFrom] to [dateTo] with period [period] and result format dates is [format] or timestamps, if format is undefined.

Examples

//  t = ["01.01.2013", "02.01.2013", "03.01.2013", "04.01.2013", "05.01.2013",
//    "06.01.2013", "07.01.2013", "08.01.2013", "09.01.2013", "10.01.2013"];
var t = tempus.generateDates({
    dateFrom: '01.01.2013',
    formatFrom: '%d.%m.%Y',
    dateTo: '10.01.2013',
    formatTo: '%d.%m.%Y',
    period: {day: 1},
    format: '%d.%m.%Y'
});
Run 
//  t = [1356998400,1357084800,1357171200,1357257600,1357344000,
//    1357430400,1357516800,1357603200,1357689600,1357776000];
var t = tempus.generateDates({
    dateFrom: '01.01.2013',
    formatFrom: '%d.%m.%Y',
    dateTo: '10.01.2013',
    formatTo: '%d.%m.%Y',
    period: {day: 1}
});
Run 
//  t = ["00:00","01:00","02:00","03:00","04:00","05:00","06:00","07:00","08:00","09:00",
//    "10:00","11:00","12:00","13:00","14:00","15:00","16:00","17:00","18:00","19:00",
//    "20:00","21:00","22:00","23:00"];
var t = tempus.generateDates({
    dateFrom: '01.01.2013 00:00',
    formatFrom: '%d.%m.%Y %H:%M',
    dateTo: '01.01.2013 23:00',
    formatTo: '%d.%m.%Y %H:%M',
    period: {hours: 1},
    format: '%H:%M'
});
Run 
//  t = ["2013-01-01","2013-02-01","2013-03-01","2013-04-01","2013-05-01",
//    "2013-06-01","2013-07-01","2013-08-01","2013-09-01","2013-10-01"];
var t = tempus.generateDates({
    dateFrom: '01.01.2013',
    formatFrom: '%d.%m.%Y',
    dateTo: '31.10.2013',

Run 
//  t = ["05.01.2013","06.01.2013","07.01.2013","08.01.2013","09.01.2013","10.01.2013","11.01.2013",
//    "12.01.2013","13.01.2013","14.01.2013","15.01.2013","16.01.2013","17.01.2013","18.01.2013",

tempus.registerFormat (value, formatFunc, parseFunc, parseLit)

Syntax

undefined registerFormat( string value, function formatFunc, function parseFunc, string parseLit )

Description

Registering a new format.

Examples

// registering a new format
tempus.registerFormat('%q',
    function(date) {
        return date.month;
    },
    function(value) {
        var v = Number(value);
        return {month: (isNaN(v) ? undefined : v) };
    },
    '\\d{1,2}'
);
// test it
// t = "01.1.2013";
var t = tempus.format({year: 2013, month: 1, day: 1}, '%d.%q.%Y');
// t = {"year":2013,"month":2,"day":10,"hours":0,"minutes":0,"seconds":0};
var t = tempus.parse('10.2.2013', '%d.%q.%Y');
Run
