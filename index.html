TempusJS demo

TempusJS is JavaScript library for most comfortable using date/time functions.

Download compressed Tempus Download uncompressed Tempus

or

$ bower install tempus

Here timestamps in seconds! In default JS in milliseconds!

Format

Directive Meaning Example
%d Day of the month as a zero-padded decimal number. 01, 02, ..., 31
%m Month as a zero-padded decimal number. 01, 02, ..., 12
%Y Year with century as a decimal number. 1970, 1988, 2001, 2013
%w Weekday as a decimal number, where 0 is Sunday and 6 is Saturday. 0, 1, ..., 6
%a Weekday abbreviated name. Sun, Mon, ..., Sat
%A Weekday full name. Sunday, Monday, ..., Saturday
%b Month abbreviated name. Jan, Feb, ..., Dec
%B Month full name. January, February, ..., December
%H Hour (24-hour clock) as a zero-padded decimal number. 00, 01, ..., 23
%M Minute as a zero-padded decimal number. 00, 01, ..., 59
%S Second as a zero-padded decimal number. 00, 01, ..., 59
%s Unix time (in seconds). 0, ..., 1377157851, ..., 2147483647
%F ISO_8601 format (%Y-%m-%d). 1970-01-01, ..., 2013-08-22, ...
%D UK and USA format (%m/%d/%Y). 01/01/1970, ..., 08/22/2013, ...

tempus.time (date, format)

Syntax

int time( [object date] )

or

int time( [string date, string format] )

Description

Returns current timestamp (UTC) in seconds. If parameter is not undefined, timestamp was received from this.

Examples

// t = 1383667313;
var t = tempus.time();
Run 
// t = 1383609600;
var t = tempus.time({year: 2013, month: 11, day: 5});
Run 
// t = 1363046400;
var t = tempus.time('2013-03-12', '%Y-%m-%d');
Run 
// t = 1363363200;
var t = tempus.time('2013-03-15 16:00', '%Y-%m-%d %H:%M');
Run

tempus.date (date)

Syntax

object date( [int timestamp|object date] )

Description

Returns date object from timestamp (UTC). Or fix received object to default tempus date object.

Examples

// t = {"year":2013,"month":11,"day":5,"hours":16,"minutes":1,"seconds":53};
var t = tempus.date(1383667313);
Run 
// t = {"year":2013,"month":11,"day":5,"hours":0,"minutes":0,"seconds":0};
var t = tempus.date(1383609600);
Run 
// t = {"year":1970,"month":1,"day":1,"hours":0,"minutes":0,"seconds":0};
var t = tempus.date({});
Run 
// t = {"year":2013,"month":10,"day":5,"hours":0,"minutes":0,"seconds":0};
var t = tempus.date({year:2013, month:10, day:5});
Run

tempus.format (date, format)

Syntax

string format( object|int date, string format )

Description

Returns formatted string of date. You can use object or timestamp as parameter of method.

tempus.between (dateFrom, dateTo, type)

Syntax

int between( object dateFrom, object dateTo, string type=day|month|year|hours|minutes|seconds )

Description

Returns integer of date between from [dateFrom] to [dateTo] as [type].

Examples

//  t = 4;
var t = tempus.between({year: 2013, month: 11, day: 1}, {year: 2013, month: 11, day: 5}, 'day');
Run 
//  t = 6;
var t = tempus.between({year: 2013, month: 11, day: 1}, {year: 2014, month: 5, day: 5}, 'month');
Run 
//  t = 266400;
var t = tempus.between({year: 2013, month: 11, day: 1}, {year: 2014, month: 5, day: 5}, 'minutes');
Run 
//  t = 10224;
var t = tempus.between({year: 2013, month: 11, day: 1}, {year: 2015, month: 1, day: 1}, 'hours');
Run 
//  Happy new year!
//  t = 56;
var t = tempus.between(tempus.now(), {year: 2014, month: 1, day: 1}, 'day');
Run

Run 
//  t = true;
var t = tempus.validate('29.02.2012', '%d.%m.%Y');
Run 
//  t = false;
var t = tempus.validate('24:61 29.02.2012', '%H:%M %d.%m.%Y');

tempus.reformat (date, formatFrom, formatTo)

Syntax

string reformat( string date, string formatFrom, string formatTo )

Description

Reformats date from one to other format.

Examples

        alert("Hello over 2 second!");
    }, 2);
Run

        alert("Hello over 0.5 second!");
    }, 0.5);
Run

tempus.setInterval (callback, timeout)

Syntax

object setInterval( function callback, int timeout )

Description

Just wrapper on setInterval, this timeout in seconds (not a milliseconds).

Examples

//  "Hello again!" 3x times;
var x = 3;
var t = tempus.setInterval(function() {
        x--;
        if (x === 0) {
            clearInterval(t);
        };
        alert("Hello again!");
    }, 2);
Run

tempus.clock (callback)

Syntax

object clock( function callback )

Description

Clock. Every second callback is calling with parameter "date". "date" - is current date object. Also run callback immediately after starting.

Examples

//  For example, please, click button
var clock = tempus.clock(function(date) {
    document.getElementById('tempus-clock-example').innerHTML = tempus.format(date, "%H:%M:%S");
});
Run

tempus.alarm (date, callback)

Syntax

object alarm( object date, function callback )

Description

Alarms at [date]. After alarming self-destructs.

Examples

//  For example, please, click button
var alarmAt = tempus.incDate(tempus.now(), 20, 'seconds');
document.getElementById('tempus-alarm-example').innerHTML = tempus.format(alarmAt, '%H:%M:%S %d.%m.%Y');
var a = tempus.alarm(alarmAt, function(date) {
    alert('Alarmed at '+tempus.format(date, '%H:%M:%S %d.%m.%Y'));
});
