tempus.generateDates (date, formatFrom, formatTo)

Syntax

string generateDates( object options )

Options:

{
    dateFrom: string|object
    formatFrom: string|undefined
    dateTo: string|object
    formatTo: string|undefined
    period: number (seconds)|object|string='seconds'|'minutes'|'hours'|'day'|'month'|'year'
    format: string|undefined,
    asObject: boolean,
    groupBy: string=year|month|week|day|hours|minutes|seconds
}

Description

Generates dates from [dateFrom] to [dateTo] with period [period] and result format dates is [format] or timestamps, if format is undefined.

Examples

Run 
//  Too long...
var t = tempus.generateDates({
    dateFrom: {year: 2013, month: 1, day: 1},
    dateTo: {year: 2013, month: 3, day: 15},
    period: {day: 1},
    format: '%d.%m.%Y',
    groupBy: 'month'
});
Run 
//  Too long...
var t = tempus.generateDates({
    dateFrom: {year: 2013, month: 1, day: 1},
    dateTo: {year: 2013, month: 1, day: 31},
    period: {day: 1},
    format: '%d.%m.%Y',
    groupBy: 'week'
});
Run

tempus.registerFormat (value, formatFunc, parseFunc, parseLit)

Syntax

undefined registerFormat( string value, function formatFunc, function parseFunc, string parseLit )

Description

Registering a new format.

Examples

// registering a new format
tempus.registerFormat('%q',
    function(date) {
        return date.month;
    },
    function(value) {
        var v = Number(value);
        return {month: (isNaN(v) ? undefined : v) };
    },
    '\\d{1,2}'
);
// test it
// t = "01.1.2013";
var t = tempus.format({year: 2013, month: 1, day: 1}, '%d.%q.%Y');
// t = {"year":2013,"month":2,"day":10,"hours":0,"minutes":0,"seconds":0};
var t = tempus.parse('10.2.2013', '%d.%q.%Y');
Run

tempus.unregisterFormat (value)

Syntax

undefined unregisterFormat( string value )

Description

Unregistering a format.

Examples

// unregistering a format
tempus.unregisterFormat('%d');
// test it
// t = "%d.01.2013";
var t = tempus.format({year: 2013, month: 1, day: 1}, '%d.%m.%Y');
